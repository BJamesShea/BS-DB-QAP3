    { title: "To Kill a Mockingbird", author: "Harper Lee", genre: "Fiction", year: 1960 },
    { title: "1984", author: "George Orwell", genre: "Dystopian", year: 1949 }
]);

// Retrieve titles of all books

// Find all books written by “J.R.R. Tolkien”

    { title: "1984" },

);
